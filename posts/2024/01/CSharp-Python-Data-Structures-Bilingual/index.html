public struct Point
{
    public int x;
    public int y;

    public Point(int x, int y)
    {
        this.x = x;
        this.y = y;
    }
}

Point p1 = new Point(1, 2);
Point p2 = new Point(3, 4);

Console.WriteLine($"p1.x = {p1.x}, p1.y = {p1.y}");
