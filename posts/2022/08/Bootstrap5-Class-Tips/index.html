route.js

function UrlRouter(){
    let hash = window.location.hash;
    switch (hash) {
        case '':
        case '#index':
            $('#post-content').load('default.html')
            break;
        default:
            $('#post-content').load(`${hash.replace('#', '')}.html`)
            break;
    }
}

$(window).on('hashchange', function(e){
    setTimeout(function () {
        window.scrollTo({top: 0, behavior: 'instant'});
    }, 100);
    UrlRouter();
});

$(document).ready(function(){
    UrlRouter();
});
