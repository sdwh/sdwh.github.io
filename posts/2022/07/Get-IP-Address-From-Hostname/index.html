Python

import socket

datas = {
'web-servernode1': '',
'web-servernode2': '',
'sql-node1': '',
}

for server in datas:
  try:
    ip_address = socket.gethostbyname(server)
  except:
    ip_address = 'unknown'
  print(f"{server}: {ip_address}")
  datas[server] = ip_address
